// Clip Ammo

actor RLClip1 : RLClip Replaces DRPGClipFull
{
	Inventory.PickupSound "items/clip"
    
    States
    {
    Spawn:
        TNT1 A 0 NoDelay ACS_NamedExecuteAlways("ItemInit", 0)
		CLIP A -1
        Stop
    }
}
